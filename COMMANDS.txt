# Command lines for building Han Peace Sans OTFs & OTCs
#
# Version 2016-03-02
#
# To subroutinize the 'CFF' table, which may take several hours for
# the 65,535-glyph fonts, replace "-nS" with "-maxs 30000" in the
# makeotf command lines.
#
# To add the "stub" 'DSIG' table to the OpenType/CFF fonts, one of the
# following AFDKO command lines can be used, depending on the directory
# in which it is executed:
#
# sfntedit -a DSIG=../DSIG.bin <font>.otf
# sfntedit -a DSIG=../../DSIG.bin <font>.otf
#
# Important Note: Do not add this 'DSIG' table to the OpenType/CFF
# fonts that are used to build the OTCs, nor to the OTCs themselves.

### Command lines for building the region-specific subset OTFs
### Executed in {ExtraLight,Light,Normal,Regular,Medium,Bold,Heavy}

# Japanese

### Command lines for building the Super OTC

# Command lines for copying the CFF from the Japanese OTF to temporary
# (tmp) copies of the Simplified Chinese, Traditional Chinese, and
# Korean OTFs (replace $dir with the face name)
# Executed in {ExtraLight,Light,Normal,Medium,Heavy}/OTC

sfntedit -x CFF=CFF.J HanPeaceSansJP-$dir.otf
cp HanPeaceSansJP-$dir.otf HanPeaceSansJP-$dir.otf.tmp
cp HanPeaceSansKR-$dir.otf HanPeaceSansKR-$dir.otf.tmp
cp HanPeaceSansSC-$dir.otf HanPeaceSansSC-$dir.otf.tmp
cp HanPeaceSansTC-$dir.otf HanPeaceSansTC-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansKR-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansSC-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansTC-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansJP-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansKR-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansSC-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansTC-$dir.otf.tmp

# Executed in {Regular,Bold}/OTC

sfntedit -x CFF=CFF.J HanPeaceSansJP-$dir.otf
cp HanPeaceSansJP-$dir.otf HanPeaceSansJP-$dir.otf.tmp
cp HanPeaceSansKR-$dir.otf HanPeaceSansKR-$dir.otf.tmp
cp HanPeaceSansSC-$dir.otf HanPeaceSansSC-$dir.otf.tmp
cp HanPeaceSansTC-$dir.otf HanPeaceSansTC-$dir.otf.tmp
cp HanPeaceSansMonoJP-$dir.otf HanPeaceSansMonoJP-$dir.otf.tmp
cp HanPeaceSansMonoKR-$dir.otf HanPeaceSansMonoKR-$dir.otf.tmp
cp HanPeaceSansMonoSC-$dir.otf HanPeaceSansMonoSC-$dir.otf.tmp
cp HanPeaceSansMonoTW-$dir.otf HanPeaceSansMonoTW-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansKR-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansSC-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansTC-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansMonoJP-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansMonoKR-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansMonoSC-$dir.otf.tmp
sfntedit -a CFF=CFF.J HanPeaceSansMonoTW-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansJP-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansKR-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansSC-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansTC-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansMonoJP-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansMonoKR-$dir.otf.tmp
sfntedit -d DSIG HanPeaceSansMonoSC-$dir.otf.tmp
